#homeassistant
sudo mkdir -p /mnt/disk/docker/ha
sudo nano compose.yaml
---------------------------------------------------------
services:
  homeassistant:
    container_name: homeassistant
    image: "ghcr.io/home-assistant/home-assistant:stable"
    volumes:
      - /mnt/disk/docker/ha:/config
      - /etc/localtime:/etc/localtime:ro
      - /run/dbus:/run/dbus:ro
    restart: unless-stopped
    privileged: true
    network_mode: bridge/host
---------------------------------------------------------------
services:
 qinglong
    container_name:qinglong
    image: whyour/qinglong:latest # 基于 Debian 的版本：whyour/qinglong:debian  
    volumes:
      - /mnt/disk/docker/ql/data:/ql/data
    ports:
      - "5700:5700"
    environment:
      QlBaseUrl: '/' # 部署路径非必须，以斜杠开头和结尾，比如 /test/
    restart: unless-stopped
--------------------------------------------------------------
#维护
cd /mnt/disk/docker/project1
sudo docker compose up -d #启动docker
sudo docker compose down #停止docker
sudo docker-compose pull #拉取最新镜像
